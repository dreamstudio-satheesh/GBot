worker_processes auto;

events {
    worker_connections 1024;
}

http {
    include       mime.types;
    default_type  application/octet-stream;

    sendfile on;
    keepalive_timeout 65;

    # Prevent issues with large file uploads
    client_max_body_size 50M;

    upstream php_backend {
        server laravel_service:9000;  # ðŸ‘ˆ Ensure this matches the container name
    }

    upstream fastapi_backend {
        server fastapi_service:8000;  # ðŸ‘ˆ Ensure this matches the container name
    }

    upstream websocket_backend {
        server websocket_service:6001;  # ðŸ‘ˆ Ensure this matches the container name
    }

    server {
        listen 80;
        server_name localhost;

        root /var/www/html/public;
        index index.php index.html index.htm;

        location / {
            try_files $uri $uri/ /index.php?$query_string;
        }

        location /api {
            proxy_pass http://fastapi_backend;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        location /ws/ {
            proxy_pass http://websocket_backend;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection "Upgrade";
        }

        location ~ \.php$ {
            include fastcgi_params;
            fastcgi_pass php_backend;
            fastcgi_index index.php;
            fastcgi_param SCRIPT_FILENAME /var/www/html$fastcgi_script_name;
        }

        location ~ /\.ht {
            deny all;
        }
    }
}
